\headerbox{Monte Carlo Simulation}{name=montecarlosimulation,column=1,row=0,aligned=introduction,above=contact}{

\textbf{Geant4 10.1-patch01}
\begin{itemize}
\item \emph{Run} beam energy for $N$ \emph{Events}
\item For each particle \emph{Track} i per Event j, tally net signal gain
  $$ g_{ij} = \left\{\begin{array}{ll}
                  q_ie^{-1}, & \text{if } q_i \rightarrow Cu \\
                  q_id_{\%}e^{-1}, & \text{if } q_i \rightarrow KA(d_{\%}) \\
                  -q_ie^{-1}, & \text{if } Cu \rightarrow q_i \\
                  -q_id_{\%}e^{-1}, & \text{if } KA(d_{\%}) \rightarrow q_i
                  \end{array}\right.
  $$
\item Charge defect $ = 1 - N^{-1} \sum_j^N \sum_i g_{ij} $
\end{itemize}
